Problem5: Reverse Integer

input: Integer x

output: Reversed digits of x; if overflow (32-bit) output 0

constraints: -2^31 â‰¤ x < 2^31


// Basic Approach 
class Solution {
    public int reverse(int x) {
        boolean isNeg = x<0?true:false;
        int arr = Math.abs(x);
        int rev = 0;
        while(arr>0){
            int temp = arr%10;
            rev = rev *10+temp;
            arr = arr/10;
        }
        return isNeg?-1*rev :rev;
    }
}
